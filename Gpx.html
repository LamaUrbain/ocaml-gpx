Module Gpx

module Gpx: sig .. end
GPX schema version 1.1

This library is based on Xml-light. Metadata and extensions are represented as Xml.xml, from Xml-light library.
type gpx = {
   version : string;
   creator : string;
   metadata : metadata option;
   wpt : wpt list;
   rte : rte list;
   trk : trk list;
   extensions : extension option;
} 
type metadata = {
   name : string option;
   desc : string option;
   author : person option;
   copyright : copyright option;
   link : link list;
   time : date_time option;
   keywords : string option;
   bounds : bounds option;
   extensions : extension option;
} 
type wpt = {
   lat : latitude;
   lon : longitude;
   time : date_time option;
   ele : float option;
   magvar : degrees option;
   geoidheight : float option;
   name : string option;
   cmt : string option;
   desc : string option;
   src : string option;
   link : link list;
   sym : string option;
   typ : string option;
   fix : fix option;
   sat : int option;
   vdop : float option;
   hdop : float option;
   pdop : float option;
   ageofdgpsdata : float option;
   dgpsid : dgps_station option;
   extensions : extension option;
} 
type rte = {
   name : string option;
   cmt : string option;
   desc : string option;
   src : string option;
   link : link list;
   number : int option;
   typ : string option;
   extensions : extension option;
   rtept : wpt list;
} 
type trk = {
   name : string option;
   cmt : string option;
   desc : string option;
   src : string option;
   link : link list;
   number : int option;
   typ : string option;
   extensions : extension option;
   trkseg : trkseg list;
} 
type extension = Xml.xml 
type trkseg = {
   trkpt : wpt list;
   extensions : extension option;
} 
type copyright = {
   author : string;
   year : int option;
   license : string option;
} 
type link = {
   href : string;
   text : string option;
   typ : string option;
} 
type email = {
   id : string;
   domain : string;
} 
type person = {
   name : string option;
   email : email option;
   link : link option;
} 
type bounds = {
   minlat : latitude;
   minlon : longitude;
   maxlat : latitude;
   maxlon : longitude;
} 
type latitude = float 
type longitude = float 
type degrees = float 
type fix =
| FIX_none
| FIX_2d
| FIX_3d
| FIX_dgps
| FIX_pps 
type dgps_station = int 
type date_time = {
   year : int;
   month : int;
   day : int;
   hour : int;
   minute : int;
   second : float;
   timezone : timezone option;
} 
type timezone =
| TIMEZONE_Z
| TIMEZONE_plus of int * int
| TIMEZONE_minus of int * int 
val of_xml : Xml.xml -> gpx
val to_xml : gpx -> Xml.xml
